```python
def process_office_pickup(account, product, quantity):
    """
    领水处理 - 兼容信用模式和预付模式
    """
    # 1. 检查账户状态
    if account.status != "active":
        raise Exception("账户状态异常")
    
    # 2. 预付模式：有余额则扣减
    if account.account_type == "prepaid" and account.remaining_qty > 0:
        use_from_prepaid(account, quantity)
        return {
            "mode": "prepaid",
            "payment_status": "paid",  # 已支付
            "settlement_status": "settled"  # 无需结算
        }
    
    # 3. 信用模式：允许欠费，后续结算
    return {
        "mode": "credit",
        "payment_status": "unpaid",  # 未支付
        "settlement_status": "pending"  # 待结算
    }
```

#### 4.3.4 结算抵扣流程

```python
def settlement_deduction(office_id, settlement_items):
    """
    结算时抵扣逻辑
    1. 先检查预付账户是否有余额
    2. 有余额则抵扣，生成抵扣记录
    3. 剩余部分走现有结算流程
    """
    account = get_office_account(office_id)
    
    if account.account_type == "prepaid" and account.remaining_qty > 0:
        # 优先扣减预付余额
        deduction_qty = min(account.remaining_qty, settlement_items.total_qty)
        account.remaining_qty -= deduction_qty
        
        # 记录抵扣流水
        record_account_transaction(
            account_id=account.id,
            type="settlement_deduction",
            quantity=-deduction_qty,
            reference_id=settlement_items.id
        )
        
        # 更新结算项目状态
        settlement_items.paid_qty = deduction_qty
        settlement_items.pending_qty = settlement_items.total_qty - deduction_qty
    
    # 剩余未抵扣部分走原结算流程
    process_original_settlement(settlement_items)
```

### 6.1 优惠计算逻辑

```python
def calculate_promotion(product, quantity, user_type="prepaid"):
    """
    优惠计算 - 仅预付用户可享
    """
    if user_type != "prepaid":
        return {"free_qty": 0, "discount_amount": 0}
    
    # 买N赠M计算
    if product.promo_type == "bulk":
        cycles = quantity // (product.promo_threshold + product.promo_gift)
        free_qty = cycles * product.promo_gift
        remainder = quantity % (product.promo_threshold + product.promo_gift)
        extra_free = remainder // product.promo_threshold * product.promo_gift
        free_qty += extra_free
    
    discount_amount = free_qty * product.price
    return {"free_qty": free_qty, "discount_amount": discount_amount}
```

def check_and_process_expired_gifts():
    """
    定时任务：检测过期赠送并处理
    """
    # 查找过期的赠送金额
    expired_gifts = db.query(GiftExpiration).filter(
        GiftExpiration.expired_date <= datetime.now(),
        GiftExpiration.status == "pending"
    ).all()
    
    for gift in expired_gifts:
        # 可选择退款或展期
        if gift.quantity > 0:
            # 生成退款记录或询问管理员
            create_refund_request(gift)
