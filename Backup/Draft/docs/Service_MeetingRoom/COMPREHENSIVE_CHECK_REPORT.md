**建议实现方案**:
```python
# 检查免费额度
def check_free_hours(office_id: int, room_id: int, month: str):
    # 查询该办公室本月已用免费时长
    used = db.query(
        "SELECT SUM(duration) FROM meeting_bookings "
        "WHERE office_id = ? AND room_id = ? "
        "AND strftime('%Y-%m', booking_date) = ? "
        "AND user_type = 'internal' AND status IN ('confirmed', 'completed')"
    )
    
    # 获取会议室免费额度
    room = db.query("SELECT free_hours_per_month FROM meeting_rooms WHERE id = ?")
    
    return room.free_hours_per_month - used
```

def update_booking_status():

    # 自动开始
    db.execute("""
        UPDATE meeting_bookings 
        SET status = 'active', updated_at = ?
        WHERE status = 'confirmed' 
        AND booking_date = ?

    """)
    
    # 自动完成

        UPDATE meeting_bookings
