#### Day 2.2: 服务可用性检查 API

**新增 API**:
```python
@router.post("/check-availability")
def check_service_availability(request: AvailabilityRequest, db: Session = Depends(get_db)):
    """
    检查服务资源可用性
    
    请求示例:
    {
        "service_id": 1,
        "time_slot": "2026-04-01 14:00-18:00",
        "office_id": 1
    }
    """
    # 检查时间冲突
    conflicting = db.query(OfficePickup).filter(
        OfficePickup.service_id == request.service_id,
        OfficePickup.office_id == request.office_id,
        OfficePickup.time_slot == request.time_slot,
        OfficePickup.settlement_status.in_(['pending', 'confirmed'])
    ).first()
    
    if conflicting:
        return {
            "available": False,
            "message": "该时间段已被预约"
        }
    
    return {
        "available": True,
        "message": "可预约"
    }
```

**测试**:

#### Day 2.3: 服务预约 API

**新增 API**:
```python
@router.post("/book")
def book_service(request: BookServiceRequest, db: Session = Depends(get_db)):
    """
    预约服务
    
    请求示例:
    {
        "office_id": 1,
        "service_id": 2,
        "quantity": 1,
        "time_slot": "2026-04-01 14:00-18:00",
        "pickup_person": "张三"
    }
    """
    # 1. 检查可用性
    availability = check_service_availability(
        AvailabilityRequest(
            service_id=request.service_id,
            time_slot=request.time_slot,
            office_id=request.office_id
        ),
        db
    )
    
    if not availability["available"]:
        raise HTTPException(status_code=400, detail=availability["message"])
    
    # 2. 创建服务记录（复用 OfficePickup 表）
    pickup = OfficePickup(
        office_id=request.office_id,
        product_id=request.service_id,  # service_id 即 product_id
        quantity=request.quantity,
        pickup_time=datetime.now(),
        time_slot=request.time_slot,
        pickup_person=request.pickup_person,
        service_type='meeting_room',  # 从服务配置获取
        settlement_status='pending'
    )
    
    db.add(pickup)
    db.commit()
    db.refresh(pickup)
    
    return {
        "success": True,
        "pickup_id": pickup.id,
        "message": "预约成功"
    }
```
