function calculatePromoGift(product, quantity) {
    if (!product.promo_threshold || !product.promo_gift) return 0;
    const cycles = Math.floor(quantity / (product.promo_threshold + product.promo_gift));
    const remainder = quantity % (product.promo_threshold + product.promo_gift);
    const extraFree = Math.floor(remainder / product.promo_threshold) * product.promo_gift;
    return cycles * product.promo_gift + extraFree;
}
```

def inventory_in(product_id: int, quantity: int, note: str = "", db: Session = Depends(get_db)):

    before_stock = product.stock
    product.stock += quantity

def recharge(account_id: int, package_id: int = None, quantity: int = None, db: Session = Depends(get_db)):
    account = db.query(OfficeAccount).filter(OfficeAccount.id == account_id).first()
    
    # 计算赠送数量
    if package_id:
        package = db.query(PrepaidPackage).filter(PrepaidPackage.id == package_id).first()
        buy_qty = package.buy_quantity
        gift_qty = package.gift_quantity
    else:
        buy_qty = quantity
        gift_qty = 0
    
    # 更新账户
    before_paid = account.paid_qty
    before_free = account.free_qty
    
    account.paid_qty += buy_qty
    account.free_qty += gift_qty
    account.total_qty = account.paid_qty + account.free_qty
    account.remaining_qty = account.total_qty
    account.account_type = "prepaid"  # 充值后转为预付模式
    
    # 记录流水

        account_id=account_id,
        type="recharge",
        quantity=buy_qty,
        quantity_type="paid",
        before_qty=before_paid,
        after_qty=account.paid_qty,
        amount=buy_qty * product.price
    )
    if gift_qty > 0:
        transaction_free = AccountTransaction(
            account_id=account_id,
            type="recharge",
            quantity=gift_qty,
            quantity_type="free",
            before_qty=before_free,

@router.post("/api/office/pickup")
def office_pickup(office_id: int, product_id: int, quantity: int, db: Session = Depends(get_db)):
    account = db.query(OfficeAccount).filter(
        OfficeAccount.office_id == office_id,
        OfficeAccount.product_id == product_id
    ).first()
    
    if account and account.account_type == "prepaid" and account.remaining_qty >= quantity:
        # 预付模式：直接扣减余额
        account.remaining_qty -= quantity
        # 记录流水
        record_pickup_transaction(account, quantity, "prepaid")
        return {"mode": "prepaid", "settlement_status": "settled"}
    else:
        # 信用模式：记录待结算
        return {"mode": "credit", "settlement_status": "pending"}

@router.post("/admin/confirm-office-settlement")
def confirm_office_settlement(office_id: int, product_id: int, db: Session = Depends(get_db)):
    account = db.query(OfficeAccount).filter(
        OfficeAccount.office_id == office_id,
        OfficeAccount.product_id == product_id
    ).first()
    
    if account and account.account_type == "prepaid" and account.remaining_qty > 0:
        # 获取待结算记录
        pending_pickups = db.query(OfficePickup).filter(
            OfficePickup.office_id == office_id,
            OfficePickup.settlement_status == "pending"
        ).all()
        
        total_pending_qty = sum(p.quantity for p in pending_pickups)
        deduction_qty = min(account.remaining_qty, total_pending_qty)
        
        # 扣减预付余额
        account.remaining_qty -= deduction_qty
        
        # 记录抵扣流水
        record_account_transaction(account, -deduction_qty, "settlement_deduction")
        
        # 更新结算记录状态
        # 部分抵扣：部分状态更新为已抵扣，部分仍待结算
        # 完全抵扣：全部状态更新为已结清
        
        return {
            "deduction_qty": deduction_qty,
            "remaining_settlement": total_pending_qty - deduction_qty
        }
    
    return {"deduction_qty": 0}
```
