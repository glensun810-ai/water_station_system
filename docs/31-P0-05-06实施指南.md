```javascript
const togglePickupSelection = (pickupId) => {
    const index = selectedPickups.value.indexOf(pickupId);
    if (index > -1) {
        selectedPickups.value.splice(index, 1);
    } else {
        selectedPickups.value.push(pickupId);
    }
};

const selectAllPickups = () => {
    if (selectedPickups.value.length === filteredPickups.value.length) {
        selectedPickups.value = [];
    } else {
        selectedPickups.value = filteredPickups.value.map(p => p.id);
    }
};

const batchApplySettlement = async () => {
    if (selectedPickups.value.length === 0) {
        alert('请选择要申请结算的记录');
        return;
    }
    
    const pendingPickups = filteredPickups.value.filter(
        p => selectedPickups.value.includes(p.id) && p.settlement_status === 'pending'
    );
    
    if (pendingPickups.length === 0) {
        alert('所选记录中没有待付款状态的记录');
        return;
    }
    
    const totalAmount = pendingPickups.reduce((sum, p) => sum + p.total_amount, 0);
    
    if (!confirm(`确定要申请结算 ${pendingPickups.length} 条记录，总金额 ¥${totalAmount.toFixed(2)} 吗？`)) {
        return;
    }
    
    try {
        const response = await fetch(`${API_BASE}/api/admin/settlement/apply`, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'Authorization': `Bearer ${token.value}`
            },
            body: JSON.stringify({
                pickup_ids: pendingPickups.map(p => p.id)
            })
        });
        
        if (!response.ok) {
            throw new Error('申请失败');
        }
        
        const result = await response.json();
        alert(result.message);
        selectedPickups.value = [];
        await loadOfficePickups();
    } catch (err) {
        console.error('Batch apply failed:', err);
        alert('批量申请失败');
    }
};

const batchConfirmSettlement = async () => {
    if (selectedPickups.value.length === 0) {
        alert('请选择要确认结算的记录');
        return;
    }
    
    const appliedPickups = filteredPickups.value.filter(
        p => selectedPickups.value.includes(p.id) && p.settlement_status === 'applied'
    );
    
    if (appliedPickups.length === 0) {
        alert('所选记录中没有待确认状态的记录');
        return;
    }
    
    const totalAmount = appliedPickups.reduce((sum, p) => sum + p.total_amount, 0);
    
    if (!confirm(`确定要确认结算 ${appliedPickups.length} 条记录，总金额 ¥${totalAmount.toFixed(2)} 吗？`)) {
        return;
    }
    
    try {
        const response = await fetch(`${API_BASE}/api/admin/settlement/batch-confirm`, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'Authorization': `Bearer ${token.value}`
            },
            body: JSON.stringify({
                pickup_ids: appliedPickups.map(p => p.id)
            })
        });
        
        if (!response.ok) {
            throw new Error('确认失败');
        }
        
        const result = await response.json();
        alert(result.message);
        selectedPickups.value = [];
        await loadOfficePickups();
    } catch (err) {
        console.error('Batch confirm failed:', err);
        alert('批量确认失败');
    }
};
```

## 四、注意事项

### 4.1 状态限制

- **批量申请**: 仅对`pending`状态记录有效
- **批量确认**: 仅对`applied`状态记录有效

### 4.2 用户反馈

- 操作前显示确认对话框
- 操作后显示成功消息
- 错误时显示友好提示

### 4.3 性能考虑

- 大量记录时考虑分批处理
- 避免一次性选择过多记录（建议<100条）
