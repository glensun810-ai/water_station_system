// Line 1289-1316
checkAuth() {
    const token = localStorage.getItem('token');
    if (!token) {
        alert('请先登录');
        window.location.href = '/portal/admin/login.html?redirect=' + encodeURIComponent(window.location.pathname);
        return false;
    }
    return true;
}

handleApiError(response) {
    if (response.status === 401) {
        localStorage.removeItem('token');
        localStorage.removeItem('userInfo');
        alert('登录已过期，请重新登录');
        window.location.href = '/portal/admin/login.html?redirect=' + encodeURIComponent(window.location.pathname);
        return true;
    }
    return false;
}

getAuthHeaders() {
    const token = localStorage.getItem('token');
    const headers = { 'Content-Type': 'application/json' };
    if (token) headers['Authorization'] = `Bearer ${token}`;
    return headers;
}
```

// Line 1035-1056
displayOffices() {
    const userRole = localStorage.getItem('user_role');
    
    // 超级管理员/管理员：显示所有办公室（按常用/不常用分类）
    if (userRole === 'admin' || userRole === 'super_admin') {
        return this.offices.filter(o => 
            this.officeTab === 'common' ? o.is_common === 1 : o.is_common !== 1
        );
    }
    
    // 普通用户/办公室管理员：显示用户所属办公室
    if (this.userOffices.length > 0) {
        const userOfficeIds = this.userOffices.map(uo => uo.id);
        return this.offices.filter(o => 
            userOfficeIds.includes(o.id) && 
            (this.officeTab === 'common' ? o.is_common === 1 : o.is_common !== 1)
        );
    }
    
    // 如果 userOffices 为空，显示后端返回的 offices（后端已过滤权限）
    return this.offices.filter(o => 
        this.officeTab === 'common' ? o.is_common === 1 : o.is_common !== 1
    );
}
```
