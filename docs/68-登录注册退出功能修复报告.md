import redis

redis_client = redis.Redis(host='localhost', port=6379, db=0)

@router.post("/logout")
def user_logout(
    credentials: HTTPAuthorizationCredentials = Depends(security),
    db: Session = Depends(get_db),
):
    """用户登出（带Token黑名单）"""
    if credentials:
        token = credentials.credentials
        payload = verify_token(token)
        
        if payload:
            # 将Token添加到黑名单，过期时间与Token一致
            expires_in = payload.get('exp') - int(datetime.now().timestamp())
            redis_client.setex(
                f"blacklist:{token}", 
                expires_in, 
                "1"
            )
            
            # 记录退出日志
            user_id = payload.get('sub')
            db.execute(
                text("""
                INSERT INTO user_login_logs 
                (user_id, login_time, login_method, status)
                VALUES (:user_id, :login_time, 'logout', 'success')
                """),
                {
                    "user_id": user_id,
                    "login_time": datetime.now().isoformat()
                }
            )
            db.commit()
    
    return {"message": "登出成功"}

#### 4. 登录失败次数限制

**问题**: 无登录失败次数限制，可能被暴力破解

**建议实现**:
```python
def check_login_attempts(username: str) -> bool:
    """检查登录失败次数"""
    key = f"login_attempts:{username}"
    attempts = redis_client.get(key)
    
    if attempts and int(attempts) >= 5:
        # 检查锁定时间
        lock_key = f"login_lock:{username}"
        if redis_client.exists(lock_key):
            return False  # 已锁定
        else:
            # 锁定过期，重置计数
            redis_client.delete(key)
            return True
    
    return True

def record_login_failure(username: str):
    """记录登录失败"""
    key = f"login_attempts:{username}"
    attempts = redis_client.incr(key)
    
    # 设置过期时间（15分钟）
    if attempts == 1:
        redis_client.expire(key, 900)
    
    # 超过5次，锁定15分钟
    if attempts >= 5:
        lock_key = f"login_lock:{username}"
        redis_client.setex(lock_key, 900, "1")
```

@router.get("/activation-status")
def check_activation_status(username: str, db: Session = Depends(get_db)):
    """查询账号激活状态"""
    query = text("SELECT id, name, is_active, created_at FROM users WHERE name = :name")
    user = db.execute(query, {"name": username}).fetchone()
    
    if not user:
        raise HTTPException(status_code=404, detail="用户不存在")
    
    return {
        "username": user.name,
        "is_active": bool(user.is_active),
        "registered_at": user.created_at,
        "message": "账号已激活" if user.is_active else "等待管理员激活"
    }
