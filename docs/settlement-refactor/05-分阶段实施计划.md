#### T2-002: 实现结算申请服务

**实现内容**:
```python
class SettlementApplyService:
    """
    结算申请服务
    """
    
    def apply_settlement(self, office_id, pickup_ids, applicant_id):
        """
        申请结算
        
        Args:
            office_id: 办公室ID
            pickup_ids: 领水记录ID列表
            applicant_id: 申请人ID
        
        Returns:
            SettlementApplication: 结算申请单
        """
        # 1. 验证申请人权限
        if not self._validate_applicant_permission(applicant_id, office_id):
            raise PermissionError("只有办公室负责人可以申请结算")
        
        # 2. 验证记录有效性
        pickups = self._validate_pickups(pickup_ids, office_id)
        
        # 3. 生成申请单编号
        application_no = self._generate_application_no()
        
        # 4. 计算总金额
        total_amount = sum(p.total_amount for p in pickups)
        
        # 5. 创建申请单
        application = SettlementApplication(
            application_no=application_no,
            office_id=office_id,
            office_name=pickups[0].office_name,
            applicant_id=applicant_id,
            applicant_name=get_user_name(applicant_id),
            record_count=len(pickup_ids),
            total_amount=total_amount,
            status='applied',
            applied_at=datetime.now()
        )
        self.db.add(application)
        self.db.flush()  # 获取application.id
        
        # 6. 创建结算明细关联
        for pickup in pickups:
            item = SettlementItem(
                application_id=application.id,
                pickup_id=pickup.id,
                pickup_time=pickup.pickup_time,
                product_name=pickup.product_name,
                quantity=pickup.quantity,
                amount=pickup.total_amount,
                pickup_status='applied'
            )
            self.db.add(item)
            
            # 7. 更新领水记录状态
            pickup.settlement_status = 'applied'
            pickup.settlement_application_id = application.id
        
        self.db.commit()
        
        # 8. 记录操作日志
        self._log_operation('apply', 'application', application.id, applicant_id)
        
        # 9. 发送通知
        self._send_notification('settlement_applied', application)
        
        return application
```

```python
def migrate_pickup_status():
    """
    迁移领水记录状态
    """
    pickups = db.query(OfficePickup).filter(OfficePickup.is_deleted == False).all()
    
    for pickup in pickups:
        # 根据旧状态设置新状态
        old_status = pickup.status or pickup.settlement_status
        
        if old_status == 'settled':
            pickup.settlement_status = 'settled'
        elif old_status == 'applied':
            pickup.settlement_status = 'applied'
            # 创建历史申请单(如果有必要)
            create_historical_application(pickup)
        else:
            pickup.settlement_status = 'pending'
    
    db.commit()
    print(f"已迁移 {len(pickups)} 条领水记录")

def create_historical_application(pickup):
    """
    为已申请的记录创建历史申请单
    """
    # 只创建申请单,不执行完整流程
    application = SettlementApplication(
        application_no=f'SA-HISTORICAL-{pickup.id}',
        office_id=pickup.office_id,
        office_name=pickup.office_name,
        applicant_id=pickup.pickup_person_id or 0,
        applicant_name=pickup.pickup_person or '历史数据',
        record_count=1,
        total_amount=pickup.total_amount,
        status='applied',
        applied_at=pickup.created_at or pickup.pickup_time,
        note='历史数据迁移'
    )
    db.add(application)
    db.flush()
    
    # 创建关联
    pickup.settlement_application_id = application.id

def validate_migration():
    """
    验证迁移结果
    """
    errors = []
    
    # 检查状态字段
    invalid_pickups = db.query(OfficePickup).filter(
        OfficePickup.settlement_status.notin_(['pending', 'applied', 'settled'])
    ).all()
    
    if invalid_pickups:
        errors.append(f"发现 {len(invalid_pickups)} 条状态无效的记录")
    
    # 检查申请单
    applications = db.query(SettlementApplication).all()
    for app in applications:
        items = db.query(SettlementItem).filter_by(application_id=app.id).all()
        if len(items) != app.record_count:
            errors.append(f"申请单 {app.id} 记录数量不一致")
    
    if errors:
        print("❌ 验证失败:")
        for error in errors:
            print(f"  - {error}")
        return False
    else:
        print("✅ 验证成功: 所有数据完整")
        return True
```

#### T3-002: 收款核验机制

**实现内容**:
```python
class PaymentVerificationService:
    """
    收款核验服务
    """
    
    def confirm_payment(self, application_id, payment_data):
        """
        确认收款(含核验)
        """
        application = self.get_application(application_id)
        
        # 1. 核对收款金额
        if payment_data.payment_amount < application.total_amount:
            # 部分收款
            return self._handle_partial_payment(application, payment_data)
        elif payment_data.payment_amount > application.total_amount:
            # 多收款项
            return self._handle_over_payment(application, payment_data)
        else:
            # 全额收款
            return self._handle_full_payment(application, payment_data)
    
    def _handle_partial_payment(self, application, payment_data):
        """
        处理部分收款
        """
        difference_amount = application.total_amount - payment_data.payment_amount
        
        application.status = 'partially_settled'
        application.payment_amount = payment_data.payment_amount
        application.difference_amount = difference_amount
        application.difference_status = 'pending'
        
        self.db.commit()
        
        # 发送通知
        self._send_notification('partial_payment', {
            'application': application,
            'difference': difference_amount
        })
        
        return {
            'status': 'partially_settled',
            'difference_amount': difference_amount,
            'options': [
                {'option': 'continue_collect', 'description': '继续追缴'},
                {'option': 'waive', 'description': '作废差额'},
                {'option': 'installment', 'description': '分期付款'}
            ]
        }
```
