## 二、详细数据表设计

### 2.1 office_pickups (领水记录表) - 修改

#### 表结构定义

```python
class OfficePickup(Base):
    """
    领水记录表(修改版)
    增加统一的结算状态字段和审计字段
    """
    __tablename__ = "office_pickups"
    
    # ========== 主键和基本信息 ==========
    id = Column(Integer, primary_key=True, index=True)
    
    # ========== 办公室信息 ==========
    office_id = Column(Integer, ForeignKey('offices.id'), nullable=False, index=True)
    office_name = Column(String(100), nullable=False)  # 冗余字段,方便查询
    office_room_number = Column(String(50), nullable=True)
    
    # ========== 产品信息 ==========
    product_id = Column(Integer, ForeignKey('products.id'), nullable=False, index=True)
    product_name = Column(String(100), nullable=False)
    product_specification = Column(String(50), nullable=True)
    
    # ========== 领取信息 ==========
    quantity = Column(Integer, nullable=False)  # 领取数量
    unit_price = Column(Float, nullable=False)  # 单价
    total_amount = Column(Float, nullable=False)  # 总金额(含优惠)
    
    # 优惠信息
    free_quantity = Column(Integer, default=0)  # 免费数量(买N赠M)
    original_amount = Column(Float, nullable=True)  # 原价金额(优惠前)
    discount_amount = Column(Float, nullable=True)  # 优惠金额
    
    # ========== 领取人信息 ==========
    pickup_person = Column(String(100), nullable=True)  # 领取人姓名
    pickup_person_id = Column(Integer, ForeignKey('users.id'), nullable=True, index=True)
    pickup_time = Column(DateTime, nullable=False, index=True)  # 领取时间
    
    # ========== 结算信息(新增) ==========
    settlement_status = Column(String(20), nullable=False, default='pending', index=True)
    # 状态定义:
    # - pending: 待付款(初始状态)
    # - applied: 已申请待审核
    # - approved: 已审核待收款
    # - confirmed: 已收款待结清
    # - settled: 已结清(终态)
    # - disputed: 有争议
    # - cancelled: 已取消/作废
    
    payment_mode = Column(String(20), nullable=True, default='credit')
    # 付款模式:
    # - credit: 记账(月结)
    # - cash: 现金
    # - transfer: 银行转账
    # - alipay: 支付宝
    # - wechat: 微信支付
    
    # ========== 结算申请关联(新增) ==========
    settlement_application_id = Column(Integer, ForeignKey('settlement_applications.id'), nullable=True, index=True)
    # 关联的结算申请单ID(当申请结算后填写)
    
    # ========== 争议信息(新增) ==========
    is_disputed = Column(Boolean, default=False)  # 是否有争议
    dispute_reason = Column(String(500), nullable=True)  # 争议原因
    dispute_evidence = Column(String(500), nullable=True)  # 争议证据URL
    dispute_reporter_id = Column(Integer, nullable=True)  # 争议报告人ID
    dispute_reported_at = Column(DateTime, nullable=True)  # 争议报告时间
    
    # ========== 备注 ==========
    note = Column(String(500), nullable=True)  # 备注
    
    # ========== 审计字段(新增) ==========
    created_at = Column(DateTime, default=datetime.now, nullable=False)
    created_by_id = Column(Integer, ForeignKey('users.id'), nullable=True)  # 创建人ID
    updated_at = Column(DateTime, default=datetime.now, onupdate=datetime.now, nullable=False)
    updated_by_id = Column(Integer, ForeignKey('users.id'), nullable=True)  # 最后修改人ID
    
    # ========== 软删除 ==========
    is_deleted = Column(Boolean, default=False, index=True)  # 软删除标记
    deleted_at = Column(DateTime, nullable=True)  # 删除时间
    deleted_by_id = Column(Integer, nullable=True)  # 删除人ID
    
    # ========== 关系定义 ==========
    office = relationship("Office", back_populates="pickups")
    product = relationship("Product", back_populates="pickups")
    pickup_person_user = relationship("User", foreign_keys=[pickup_person_id])
    settlement_application = relationship("SettlementApplication", back_populates="items")
```

#### 索引设计

### 2.2 settlement_applications (结算申请单表) - 新增

#### 表结构定义

```python
class SettlementApplication(Base):
    """
    结算申请单表
    记录每次结算申请的完整信息
    """
    __tablename__ = "settlement_applications"
    
    # ========== 主键和编号 ==========
    id = Column(Integer, primary_key=True, index=True)
    application_no = Column(String(50), unique=True, nullable=False, index=True)
    # 编号规则: SA-YYYYMMDD-NNNN (Settlement Application)
    # 示例: SA-20260408-0001
    
    # ========== 办公室信息 ==========
    office_id = Column(Integer, ForeignKey('offices.id'), nullable=False, index=True)
    office_name = Column(String(100), nullable=False)
    office_room_number = Column(String(50), nullable=True)
    
    # ========== 申请信息 ==========
    applicant_id = Column(Integer, ForeignKey('users.id'), nullable=False, index=True)
    applicant_name = Column(String(100), nullable=False)
    applied_at = Column(DateTime, nullable=False, default=datetime.now, index=True)
    
    # ========== 结算信息 ==========
    record_count = Column(Integer, nullable=False)  # 关联记录数量
    total_amount = Column(Float, nullable=False)  # 申请总金额
    
    # ========== 状态管理 ==========
    status = Column(String(20), nullable=False, default='applied', index=True)
    # 状态定义:
    # - draft: 草稿(未提交)
    # - applied: 已申请待审核
    # - approved: 已审核待收款
    # - confirmed: 已收款待结清
    # - settled: 已结清(终态)
    # - rejected: 已拒绝
    # - cancelled: 已取消
    # - partially_settled: 部分结清
    
    # ========== 审核信息 ==========
    reviewer_id = Column(Integer, ForeignKey('users.id'), nullable=True)
    reviewer_name = Column(String(100), nullable=True)
    reviewed_at = Column(DateTime, nullable=True)
    review_status = Column(String(20), nullable=True)  # approved/rejected
    review_note = Column(String(500), nullable=True)  # 审核备注
    
    # ========== 收款信息 ==========
    confirmer_id = Column(Integer, ForeignKey('users.id'), nullable=True)
    confirmer_name = Column(String(100), nullable=True)
    confirmed_at = Column(DateTime, nullable=True)
    
    payment_method = Column(String(20), nullable=True)
    # 收款方式: cash/transfer/alipay/wechat/credit
    
    payment_amount = Column(Float, nullable=True)  # 实际收款金额
    payment_time = Column(DateTime, nullable=True)  # 收款时间
    payment_evidence = Column(String(500), nullable=True)  # 收款凭证URL
    
    # ========== 差额处理(新增) ==========
    difference_amount = Column(Float, nullable=True)  # 差额金额(申请金额-实际收款)
    difference_status = Column(String(20), nullable=True)  # pending/waived/installment
    difference_note = Column(String(500), nullable=True)  # 差额说明
    
    # ========== 结算周期 ==========
    settlement_period = Column(String(20), nullable=True)  # 结算周期(如: 2026-04)
    settlement_type = Column(String(20), nullable=True, default='adhoc')
    # 结算类型:
    # - monthly: 月度结算
    # - adhoc: 临时结算
    
    # ========== 月度结算关联 ==========
    monthly_settlement_id = Column(Integer, ForeignKey('monthly_settlements.id'), nullable=True)
    # 关联的月度结算单ID
    
    # ========== 收据信息 ==========
    receipt_no = Column(String(50), nullable=True, unique=True)  # 收据编号
    receipt_generated_at = Column(DateTime, nullable=True)  # 收据生成时间
    
    # ========== 备注 ==========
    note = Column(String(500), nullable=True)  # 申请备注
    
    # ========== 审计字段 ==========
    created_at = Column(DateTime, default=datetime.now, nullable=False)
    created_by_id = Column(Integer, ForeignKey('users.id'), nullable=True)
    updated_at = Column(DateTime, default=datetime.now, onupdate=datetime.now, nullable=False)
    updated_by_id = Column(Integer, ForeignKey('users.id'), nullable=True)
    
    # ========== 软删除 ==========
    is_deleted = Column(Boolean, default=False, index=True)
    deleted_at = Column(DateTime, nullable=True)
    deleted_by_id = Column(Integer, nullable=True)
    
    # ========== 关系定义 ==========
    office = relationship("Office", back_populates="settlement_applications")
    applicant = relationship("User", foreign_keys=[applicant_id])
    reviewer = relationship("User", foreign_keys=[reviewer_id])
    confirmer = relationship("User", foreign_keys=[confirmer_id])
    items = relationship("SettlementItem", back_populates="application", cascade="all, delete-orphan")
    monthly_settlement = relationship("MonthlySettlement", back_populates="applications")
```

```python
class SettlementLog(Base):
    """
    结算操作日志表
    记录所有结算相关操作,用于审计追溯
    """
    __tablename__ = "settlement_logs"
    
    # ========== 主键 ==========
    id = Column(Integer, primary_key=True, index=True)
    
    # ========== 操作信息 ==========
    operation_type = Column(String(50), nullable=False, index=True)
    # 操作类型:
    # - apply: 申请结算
    # - approve: 审核通过
    # - reject: 审核拒绝
    # - confirm: 确认收款
    # - cancel: 取消结算
    # - dispute: 报告争议
    # - resolve_dispute: 解决争议
    # - waive_difference: 作废差额
    # - force_settle: 强制结清
    
    operation_status = Column(String(20), nullable=False)
    # 操作状态: success/failed
    
    # ========== 操作对象 ==========
    # 可以是领水记录或结算申请单
    target_type = Column(String(20), nullable=False)
    # target类型: pickup/application/monthly_settlement
    
    target_id = Column(Integer, nullable=False, index=True)
    # 目标对象ID
    
    # ========== 状态变更记录 ==========
    old_status = Column(String(20), nullable=True)  # 原状态
    new_status = Column(String(20), nullable=True)  # 新状态
    
    # ========== 操作人信息 ==========
    operator_id = Column(Integer, ForeignKey('users.id'), nullable=False, index=True)
    operator_name = Column(String(100), nullable=False)
    operator_role = Column(String(20), nullable=False)  # admin/office_manager/user
    
    # ========== 操作详情 ==========
    operation_detail = Column(Text, nullable=True)  # 操作详情(JSON格式)
    # 示例:

    #   "pickup_ids": [1, 2, 3],
    #   "total_amount": 856.00,
    #   "payment_method": "transfer",
    #   "payment_evidence": "https://..."
    # }
    
    # ========== 备注 ==========
```

```python
class MonthlySettlement(Base):
    """
    月度结算单表
    记录按月汇总的结算信息
    """
    __tablename__ = "monthly_settlements"
    
    # ========== 主键和编号 ==========
    id = Column(Integer, primary_key=True, index=True)
    settlement_no = Column(String(50), unique=True, nullable=False, index=True)
    # 编号规则: MS-YYYYMM-NNNN (Monthly Settlement)
    # 示例: MS-202604-0001
    
    # ========== 结算周期 ==========
    settlement_period = Column(String(20), nullable=False, unique=True, index=True)
    # 结算周期: YYYY-MM格式
    # 示例: 2026-04
    
    settlement_year = Column(Integer, nullable=False, index=True)  # 年份
    settlement_month = Column(Integer, nullable=False, index=True)  # 月份
    
    start_date = Column(DateTime, nullable=False)  # 开始日期(月初)
    end_date = Column(DateTime, nullable=False)  # 结束日期(月末)
    
    # ========== 汇总信息 ==========
    office_count = Column(Integer, nullable=False)  # 涉及办公室数量
    record_count = Column(Integer, nullable=False)  # 领水记录总数量
    total_amount = Column(Float, nullable=False)  # 结算总金额
    
    # ========== 办公室明细(JSON) ==========
    office_details = Column(Text, nullable=True)  # JSON格式
    # 示例:
    # [
    #   {
    #     "office_id": 1,
    #     "office_name": "A101办公室",
    #     "record_count": 23,
    #     "amount": 856.00,
    #     "status": "settled",
    #     "settlement_id": 101
    #   },
    #   ...
    # ]
    
    # ========== 状态管理 ==========
    status = Column(String(20), nullable=False, default='draft', index=True)
    # 状态定义:
    # - draft: 草稿(未审核)
    # - pending_payment: 待付款(审核通过)
    # - partially_settled: 部分结清
    # - settled: 已结清(终态)
    # - cancelled: 已取消
    
    # ========== 统计信息 ==========
    settled_office_count = Column(Integer, default=0)  # 已结清办公室数量
    unsettled_office_count = Column(Integer, default=0)  # 未结清办公室数量
    settled_amount = Column(Float, default=0)  # 已结清金额
    unsettled_amount = Column(Float, default=0)  # 未结清金额
    
    # ========== 创建信息 ==========
    created_at = Column(DateTime, default=datetime.now, nullable=False)
    created_by_id = Column(Integer, ForeignKey('users.id'), nullable=False)
    created_type = Column(String(20), nullable=False)  # auto/manual
    
    # ========== 完成信息 ==========
    completed_at = Column(DateTime, nullable=True)  # 完成时间
    
    # ========== 备注 ==========
    note = Column(String(500), nullable=True)  # 备注
    
    # ========== 审计字段 ==========
    updated_at = Column(DateTime, default=datetime.now, onupdate=datetime.now)
    is_deleted = Column(Boolean, default=False, index=True)
    
    # ========== 关系定义 ==========
    applications = relationship("SettlementApplication", back_populates="monthly_settlement")
    created_by = relationship("User", foreign_keys=[created_by_id])
```

## 三、数据表关系设计

### 3.1 外键关系图

```
office_pickups
  ├─ office_id → offices.id (N:1)
  ├─ product_id → products.id (N:1)
  ├─ pickup_person_id → users.id (N:1)
  ├─ settlement_application_id → settlement_applications.id (N:1)
  
settlement_applications
  ├─ office_id → offices.id (N:1)
  ├─ applicant_id → users.id (N:1)
  ├─ reviewer_id → users.id (N:1)
  ├─ confirmer_id → users.id (N:1)
  ├─ monthly_settlement_id → monthly_settlements.id (N:1)
  
settlement_items
  ├─ application_id → settlement_applications.id (N:1)
  ├─ pickup_id → office_pickups.id (N:1)
  
settlement_logs
  └─ operator_id → users.id (N:1)
  
monthly_settlements
  ├─ created_by_id → users.id (N:1)
  └─ applications → settlement_applications (1:N)
```

---

### 3.2 关系约束

#### 约束1: 领水记录与结算申请单关系
**规则**: 一条领水记录只能属于一个结算申请单
**实现**: 
- `office_pickups.settlement_application_id` 唯一索引
- `settlement_items` 表的 (application_id, pickup_id) 唯一约束

---

#### 约束2: 结算申请单与月度结算单关系
**规则**: 一个结算申请单只能属于一个月度结算单
**实现**: 
- `settlement_applications.monthly_settlement_id` 外键约束
- 当月度结算单删除时,关联的申请单不受影响(保留)

---

#### 约束3: 状态一致性约束
**规则**: 领水记录状态必须与所属结算申请单状态一致
**实现**: 
- 应用层校验:修改状态时同步更新
- 数据库触发器(可选):自动同步状态

#### 步骤2: 数据清洗和状态迁移

```python
"""
数据迁移脚本: 迁移现有领水记录的状态字段
"""
from datetime import datetime

def migrate_pickup_status(db):
    """
    迁移领水记录状态
    将旧的状态字段映射到新的settlement_status字段
    """
    
    # 查询所有领水记录
    pickups = db.query(OfficePickup).all()
    
    for pickup in pickups:
        # 状态映射规则
        old_status = getattr(pickup, 'status', None) or getattr(pickup, 'settlement_status', None)
        
        if old_status == 'settled':
            # 已结清记录,保持settled状态
            pickup.settlement_status = 'settled'
            
        elif old_status == 'applied':
            # 已申请记录,需要创建对应的结算申请单(历史数据)
            pickup.settlement_status = 'applied'
            
            # 创建历史结算申请单(可选)
            # application = create_historical_application(pickup)
            # pickup.settlement_application_id = application.id
            
        elif old_status == 'pending' or old_status is None:
            # 待付款记录,设置为pending
            pickup.settlement_status = 'pending'
            
        else:
            # 其他状态,默认设置为pending
            pickup.settlement_status = 'pending'
        
        # 设置其他新字段默认值
        if not pickup.payment_mode:
            pickup.payment_mode = 'credit'
        
        if not pickup.is_deleted:
            pickup.is_deleted = False
        
        if not pickup.is_disputed:
            pickup.is_disputed = False
    
    db.commit()
    
    print(f"迁移完成: 共处理 {len(pickups)} 条记录")
```

### 4.4 数据验证脚本

```python
"""
数据迁移验证脚本
验证迁移后的数据完整性和一致性
"""

def validate_migration(db):
    """
    验证数据迁移结果
    """
    
    errors = []
    
    # 1. 验证领水记录状态字段
    pickups = db.query(OfficePickup).filter(OfficePickup.is_deleted == False).all()
    
    for pickup in pickups:
        # 检查状态字段是否有效
        valid_statuses = ['pending', 'applied', 'approved', 'confirmed', 'settled', 'disputed', 'cancelled']
        if pickup.settlement_status not in valid_statuses:
            errors.append(f"领水记录 {pickup.id} 状态无效: {pickup.settlement_status}")
        
        # 检查是否有结算申请单关联但状态不一致
        if pickup.settlement_application_id:
            application = db.query(SettlementApplication).filter_by(id=pickup.settlement_application_id).first()
            if application and pickup.settlement_status != application.status:
                errors.append(f"领水记录 {pickup.id} 与申请单 {application.id} 状态不一致")
    
    # 2. 验证结算申请单数据
    applications = db.query(SettlementApplication).filter(SettlementApplication.is_deleted == False).all()
    
    for app in applications:
        # 检查申请单编号格式
        if not app.application_no.startswith('SA-'):
            errors.append(f"申请单 {app.id} 编号格式错误: {app.application_no}")
        
        # 检查申请单与明细记录的一致性
        items = db.query(SettlementItem).filter_by(application_id=app.id).all()
        
        if len(items) != app.record_count:
            errors.append(f"申请单 {app.id} 记录数量不一致: 申请单={app.record_count}, 实际={len(items)}")
        
        # 检查金额计算
        total_from_items = sum(item.amount for item in items)
        if abs(total_from_items - app.total_amount) > 0.01:
            errors.append(f"申请单 {app.id} 金额不一致: 申请单={app.total_amount}, 明细={total_from_items}")
    
    # 3. 验证月度结算单数据
    monthly_settlements = db.query(MonthlySettlement).filter(MonthlySettlement.is_deleted == False).all()
    
    for ms in monthly_settlements:
        # 检查编号格式
        if not ms.settlement_no.startswith('MS-'):
            errors.append(f"月度结算单 {ms.id} 编号格式错误: {ms.settlement_no}")
        
        # 检查结算周期格式

            errors.append(f"月度结算单 {ms.id} 结算周期格式错误: {ms.settlement_period}")
    
    # 输出验证结果
    if errors:
        print("❌ 数据迁移验证失败:")
        for error in errors:
            print(f"  - {error}")
        return False
    else:
        print("✅ 数据迁移验证成功: 所有数据完整且一致")
```
